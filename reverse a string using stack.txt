#include <stdio.h>
#include <string.h>
#define SIZE 100

char stack[SIZE];
int top = -1;

void push(char c) {
    if (top < SIZE - 1)
        stack[++top] = c;
}

char pop() {
    if (top >= 0)
        return stack[top--];
    return '\0';
}

int main() {
    char str[SIZE], reversed[SIZE];
    int i;

    printf("Enter a string: ");
    gets(str);

    for (i = 0; str[i] != '\0'; i++)
        push(str[i]);

    for (i = 0; str[i] != '\0'; i++)
        reversed[i] = pop();

    reversed[i] = '\0';

    printf("Reversed string: %s\n", reversed);

    return 0;
}
